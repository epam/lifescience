#####################
Deconvolution options
#####################

.. indigo_option::
    :name: deconvolution-aromatization
    :type: boolean
    :default: true
    :short: Aromatize input molecules


.. indigo_option::
    :name: deco-save-ap-bond-orders
    :type: boolean
    :default: false
    :short: Save attachement points as separate atoms with explicit bond connected.

.. indigo_option::
    :name: deco-ignore-errors
    :type: boolean
    :default: true
    :short: Ignore errors during decomposition
